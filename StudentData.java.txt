import java.util.Scanner;
class Student{
String student_name;
double student_grade;
int student_id;


Student(String student_name,double student_grade,int student_id){
this.student_name= student_name; 
this.student_grade=student_grade;
this.student_id=student_id;
}
void displayInfo(){System.out.println("student_name:" +  student_name  +  "student_grade:"  +  student_grade  +  "student_id:" + student_id);
}
}
public class StudentData{
public static void main(String[] args){

{

 input.nextLine();

 input.nextLine();
